```kotlin showLineNumbers
fun main() {
    val a = 5
    val b = 10
    val resultat = calcul(a, b)
    println(resultat)
}

fun calcul(x: Int, y: Int): Int {
    val z = x + y
    val zz = double(z)
    return zz
}

fun double(n: Int): Int {
    val nombreDouble = n * 2
    return nombreDouble
}
```
